# 🌲Задания. Структура списка  (ID в таблице b_iblock_element_property)
> [!note] ID списка в таблице IBLOCK_ID в таблице b_iblock_element = **76**
>  


- & В основную таблицу `b_iblock_element` **заполняем** стандартные поля как:
	- `IBLOCK_ID` = **76**
	- `NAME` = Тип лимита
		- Базовый
		- К решений
		- К планера
	- `DATE_CREATE` = NOW()
		- Пишем дату создания записи
	- `ACTIVE` = 'Y'
		- **Флаг активности** ('Y'/'N'). Можно временно скрывать элементы без удаления.
- & Забираем ключевое поле `ID`, чтобы связывать таблицу пользовательских полей с материнской
- & И дальше в польз таблицу пишем нужную информацию 
	- ~ `ID (IBLOCK_PROPERTY_ID)` в таблице `b_iblock_element_property`


> [!important] `NAME` Тип лимита из базовой таблицы
>  strLimType

``` sql
SELECT * FROM b_lists_field WHERE IBLOCK_ID = 76;
```

## 💾2Б Обновление базового лимита в Списке

```php
// === Подготовительный блок ==========
// === Результат запроса в переменную ===
function getSQLsingle($sql, $field, $DB) {
    $rs = $DB->Query($sql);
    if ($ar = $rs->Fetch()) {
        return $ar[$field] ?? null;
    }
    return null;
}

// === Выполнение SQL-запроса без возврата ===
function executeSQL($sql, $DB, $description = '') {
    try {
        $DB->Query($sql);
        return true;
    } catch (Exception $e) {
        return false;
    }
}

// === ID смарта ================================
$docId = $this->GetDocumentId(); 
$fullId = $docId[2]; 
$parts = explode('_', $fullId);
$smartItemId = (int)end($parts);

// === Входные параметры ======
global $DB;

// Получаем новое значение лимита из переменной БП
$intLimValue = (int)$this->GetVariable('intLimBaseValue'); // новое значение
$strPMcomment = $this->GetVariable('strPMcomment'); // коммент


// ============================================
// 1. Находим ID записи с базовым лимитом
// ============================================
$findSql = "SELECT e.ID as element_id
            FROM b_iblock_element e
            INNER JOIN b_iblock_element_property p331 ON e.ID = p331.IBLOCK_ELEMENT_ID 
                AND p331.IBLOCK_PROPERTY_ID = 331
            WHERE e.IBLOCK_ID = 76
                AND e.NAME = 'Базовый'
                AND p331.VALUE = '" . $DB->ForSql($smartItemId) . "'";

$elementId = getSQLsingle($findSql, 'element_id', $DB);

if (!$elementId) {
    $this->WriteToTrackingService("❌ Не найдена запись базового лимита для смарт-процесса " . $smartItemId, 0, 4);
    $this->SetVariable('intLimBaseValue', 0);
    return true;
}


// ============================================
// 2. Обновляем значение лимита (PROPERTY_336)
// ============================================
$updateSql = "UPDATE b_iblock_element_property 
              SET VALUE = '" . $DB->ForSql($intLimValue) . "',
                  VALUE_NUM = " . $intLimValue . "
              WHERE IBLOCK_ELEMENT_ID = " . $elementId . " 
              AND IBLOCK_PROPERTY_ID = 336";

$updateResult = executeSQL($updateSql, $DB);

if ($updateResult) {
    $this->WriteToTrackingService("✅ Значение лимита обновлено на: " . $intLimValue, 0, 6);
} else {
    $this->WriteToTrackingService("❌ Ошибка при обновлении лимита", 0, 4);
}

// ============================================
// 3. Пишем коммент РП (PROPERTY_343)
// ============================================
if (!empty($strPMcomment)) {
    $updateCommentSql = "UPDATE b_iblock_element_property 
                        SET VALUE = '" . $DB->ForSql($strPMcomment) . "'
                        WHERE IBLOCK_ELEMENT_ID = " . $elementId . " 
                        AND IBLOCK_PROPERTY_ID = 343";
    executeSQL($updateCommentSql, $DB);    
}

// ============================================
// 4. Сбрасываем кэш элемента списка
// ============================================
$DB->Query("UPDATE b_iblock_element SET TIMESTAMP_X = NOW() WHERE ID = " . $elementId);


return true;
```
