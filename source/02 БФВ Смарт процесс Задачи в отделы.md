# 📜 Логика процесса

## 🅰 Создание СП и привязка его к сделке
- & Предыстория
	- СП создается через БП в сделке
	- Параметры БП
		- Выбираются вручную при запуске
			- Название Задачи (>> Смарт процесса)
			- РО
		- Берутся автоматически
			- РП берется из карточки БП


``` php
// Подключаем класс ItemIdentifier для создания идентификаторов элементов CRM.
use Bitrix\Crm\ItemIdentifier;
// Подключаем класс Container для связи Сделки и Смарт процесса
use Bitrix\Crm\Service\Container;

// Работаем с БД
global $DB;

// Забираем входные параметры для СП
// Название Смарт процесса
$title = $this->GetVariable('strTitle');
// ID сделки 
$dealId = (int)$this->GetVariable('intDeal');
// Ответственный на смарт-процесс = РП
$responsibleId = (int)$this->GetVariable('intHoPr');
// Руководитель отдела
$HeadOfDep = (int)$this->GetVariable('intHoD');
// Код объекта
$strObjCode = $this->GetVariable('strObjCode');
// Краткое наименование объекта
$strObjShName = $this->GetVariable('strObjShName');
// Дополнительные роли
// РО РП
$intHODPM = $this->GetVariable('intHODPM');
// ГС 5
$intCSTD = $this->GetVariable('intCSTD');
// РГП
$intHOGOP = $this->GetVariable('intHOGOP');
// Этап
$strMST = $this->GetVariable('strMST');

// ======================================
// Знаем, что ID нашего смарт процесса = 1036
// Создаем смарт процесс 
$factory = Container::getInstance()->getFactory(1036);
$item = $factory->createItem();

// Наполняем его данными
// Сразу пишем стадию в поле UF_CRM_6_FROZEN_STAGE
// Метод принимает массив с полями
$item->setFromCompatibleData([
    'TITLE' => $title,
    'ASSIGNED_BY_ID' => $responsibleId,
    'UF_CRM_6_HEAD_OF_DEP' => $HeadOfDep,
    'UF_CRM_6_FROZEN_STAGE' => "DT1036_11:NEW",
    'UF_CRM_6_OBJ_CODE' => $strObjCode,
    'UF_CRM_6_OBJ_SH_NAME' => $strObjShName,    
    'UF_CRM_6_HOD_PM' => $intHODPM,
    'UF_CRM_6_CS_TD' => $intCSTD,
    'UF_CRM_6_HOG_OP' => $intHOGOP,
    'UF_CRM_6_MST' => $strMST
]);

//Сохраняет элемент в БД и идет дальше, если успешно
if ($item->save()->isSuccess()) { 
	// Забираем ID созданного элемента в БД и делаем привязку к сделке
    $smartItemId = $item->getId();
    Container::getInstance()->getRelationManager()->bindItems(
        new ItemIdentifier(CCrmOwnerType::Deal, $dealId),
        new ItemIdentifier(1036, $smartItemId)
    );
}

//Выгружаем в переменную БП ID, чтобы запускать БП для нового смарта
$this->SetVariable('intSmartItemID', $smartItemId);

// Финиш
return true;
```
